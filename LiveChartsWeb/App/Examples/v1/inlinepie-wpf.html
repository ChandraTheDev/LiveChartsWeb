In Line Syntax

Fast way to build a chart only with XAML 

<lvc:PieChart>
  <lvc:PieChart.Series>
    <lvc:PieSeries Values="3" />
    <lvc:PieSeries Values="5" />
    <lvc:PieSeries Values="2" />
    <lvc:PieSeries Values="4" />
  </lvc:PieChart.Series>
</lvc:PieChart>
