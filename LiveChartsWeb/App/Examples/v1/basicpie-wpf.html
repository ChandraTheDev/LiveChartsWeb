Basic Pie

In this example we are building a basic chart with the default configuration, this chart is able to update automatically when data or chart size changes.

Create a new Window, replace Window1.cs with he next code 

using System.Windows;
using LiveCharts;
namespace LiveChartsTest
{
  public partial class Window1
  {
   
    public Window1()
    {
      InitializeComponent();
      //we create a new SeriesCollection
      Series = new SeriesCollection();

      //create some series
      var charlesSeries = new PieSeries
      {
        Title = "Charles",
        Values = new ChartValues<double> {10}
      };
      var jamesSeries = new PieSeries
      {
        Title = "James",
        Values = new ChartValues<double> {5}
      };
      var mariaSeries = new PieSeries
      {
        Title = "James",
        Values = new ChartValues<double> {7}
      };

      //add series to SeriesCollection
      Series.Add(charlesSeries);
      Series.Add(jamesSeries);
      Series.Add(mariaSeries);

      //that's it, LiveCharts is ready and listening for your data changes.
      DataContext = this;
    }

    public SeriesCollection Series { get; set; }

  }
}

Now add a new chart to your Window1 XAML 

<lvc:PieChart LegendLocation="Right" Series="{Binding Series}" >
  <lvc:PieChart.AxisY>
    <lvc:Axis Title="Sold Items" />
  </lvc:PieChart.AxisY>
  <lvc:PieChart.AxisX>
    <lvc:Axis Title="Month"
              Labels="Jan, Feb , Mar, Apr, May, Jun" />
  </lvc:PieChart.AxisX>
</lvc:PieChart>

That’s it now you can add more series, or more values to a series and chart will automatically update: 

//Add Points
Series[0].Values.Add(10);
//Remove Points
Series[0].Values.RemoveAt(0);

//Add or remove series too
var newSeries = new BarSeries { Values = new ChartValues<double> {1,4,2} };
Series.Add(newSeries);
Series.RemoveAt(0);
