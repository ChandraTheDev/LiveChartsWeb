Inverted Charts

You can easly invert any chart type, this example is for a BarChart but it should be work with any type.

Add a new Windo to your solution and replace Window1.cs with the following code: 

using System.Windows;
using System.Windows.Media;
using LiveCharts;
namespace ChartsTest.BarExamples.RotatedBar
{

  public partial class RotatedBar
  {
    public RotatedBar()
    {
      InitializeComponent();

      var config = new SeriesConfiguration<double>()
                      .X(value => value); //scale on X
                                          //do not configure Y so it uses default config, a zero based index
      SeriesCollection =
        new SeriesCollection(config)
        {
          new BarSeries
          {
            Title = "inverted series",
            Values = new double[] {10, 15, 18, 20, 15, 13}.AsChartValues(),
            DataLabels = true
          },
          new BarSeries
          {
            Title = "inverted series 2",
            Values = new double[] {4, 8, 19, 19, 16, 12}.AsChartValues(),
            DataLabels = true
          },
          new LineSeries
          {
            Title = "inverted line series",
            Values = new double[] {10, 15, 18, 20, 15, 13}.AsChartValues(),
            Fill = Brushes.Transparent
          }
        };
       
      DataContext = this;
    }
      
    public SeriesCollection SeriesCollection { get; set; }
    
  }
}

Add the chart to your window XAML, notice it is really important to set the Invert property to true. 

<lvc:BarChart Invert="True" Series="{Binding SeriesCollection}" LegendLocation="Top">
  <lvc:BarChart.AxisY>
    <lvc:Axis Labels="Day 1, Day 2, Day 3, day 4, day5, day6, day 7"></lvc:Axis>
  </lvc:BarChart.AxisY>
</lvc:BarChart>
