In line charting

LiveCharts is designed to be easy, and of couse there are sometimes that you only need a chart and you don't care about constantly changes, all you need is to display a small and static amount of data, this library is ready to plot now, just with XAML 

<lvc:BarChart Height="300">
  <lvc:BarChart.Series>
    <lvc:BarSeries Title="Maria" Values="20, 40, 45, 60" />
    <lvc:BarSeries Title="John" Values="30, 35, 43, 68" />
    <lvc:LineSeries Title="Averge" Values="25, 37.5, 44, 64" Fill="Transparent" />
  </lvc:BarChart.Series>
</lvc:BarChart>

<lvc:LineChart Height="300">
  <lvc:LineChart.Series>
    <lvc:LineSeries Values="25, 37.5, 44, 64" Fill="Transparent" />
  </lvc:LineChart.Series>
</lvc:LineChart>

<lvc:PieChart Height="300">
  <lvc:PieChart.Series>
    <lvc:PieSeries Title="Chrome" Values="25" />
    <lvc:PieSeries Title="Mozilla" Values="37.5"/>
    <lvc:PieSeries Title="Explorer" Values="13"/>
  </lvc:PieChart.Series>
</lvc:PieChart>
