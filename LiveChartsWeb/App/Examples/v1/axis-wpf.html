Styling Axes

The Axis class gathers all the information about how to draw the set of perpendicular lines and labels to an axis. 

<lvc:BarChart>
    <lvc:BarChart.AxisY>
        <lvc:Axis Title="Sold Items Change" IsEnabled="True" Color="LimeGreen" 
                StrokeThickness="4" Foreground="MediumSeaGreen" FontSize="13" FontFamily="Arial">
            <lvc:Axis.Separator>
                <lvc:Separator Color="ForestGreen" Thickness="2" IsEnabled="True"></lvc:Separator>
            </lvc:Axis.Separator>
        </lvc:Axis>
    </lvc:BarChart.AxisY>
    <lvc:BarChart.AxisX>
        <lvc:Axis Title="Month" 
                Labels="Jan, Feb, Mar, Apr, May, Jun, Ago, Sep, Oct, Nov, Dec" 
                IsEnabled="True"
                Color="CornflowerBlue"
                Foreground="DeepSkyBlue"
                StrokeThickness="4">
            <lvc:Axis.Separator>
                <lvc:Separator Color="DodgerBlue" Thickness="2" IsEnabled="True" ></lvc:Separator>
            </lvc:Axis.Separator>
        </lvc:Axis>
    </lvc:BarChart.AxisX>
</lvc:BarChart>
