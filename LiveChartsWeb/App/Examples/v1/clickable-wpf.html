Clickable Chart Points

You can use the DataClick event to fire any action.

XAML 

<lvc:LineChart DataClick="Chart_OnDataClick" />
<lvc:BarChart DataClick="Chart_OnDataClick" />
<lvc:StackedBarChart DataClick="Chart_OnDataClick" />
<lvc:PieChart DataClick="Chart_OnDataClick" />

Code Behind 

private void Chart_OnDataClick(ChartPoint point)
{
  //DataClick event will be called when a user clicks on a data point.
           
  //point.Instance is normally double if you did not change default configuration
  //in this case we configured our series to use MonthSalesData class.
  var salesData = point.Instance as MonthSalesData;
  if (salesData == null) return;
  MessageBox.Show("You clicked on: (" + point.X + ", " + point.Y + "), " +
                         "sold items:" + salesData.SoldItems + ", " +
                         "best sellers: " + salesData.BestSellers.Aggregate((x, y) => x + ", " + y));
}
