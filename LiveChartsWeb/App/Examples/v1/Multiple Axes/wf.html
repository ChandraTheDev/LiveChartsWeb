Multiple axes

You can add as many axes as you need, every axes scales independently with its own series, to add a series to an axis you must specify it using the Series.ScalesXAt or Series.ScalesYAt the default value of both properties is zero.

Code Behind

using System.Windows.Forms;
using LiveCharts;
using LiveCharts.Wpf;
namespace Winforms.Cartesian.MultiAxes
{
    public partial class MultipleAxesExample : Form
    {
        public MultipleAxesExample()
        {
            InitializeComponent();
            cartesianChart1.Series.Add(new LineSeries
            {
                Values = new ChartValues<double> {1, 5, 3, 5, 3},
                ScalesYAt = 0
            });
            cartesianChart1.Series.Add(new LineSeries
            {
                Values = new ChartValues<double> { 20, 30, 70, 20, 10 },
                ScalesYAt = 1
            });
            cartesianChart1.Series.Add(new LineSeries
            {
                Values = new ChartValues<double> { 600, 300, 200, 600, 800 },
                ScalesYAt = 2
            });
            //now we add the 3 axes
            cartesianChart1.AxisY.Add(new Axis
            {
                Foreground = System.Windows.Media.Brushes.DodgerBlue,
                Title = "Blue Axis"
            });
            cartesianChart1.AxisY.Add(new Axis
            {
                Foreground = System.Windows.Media.Brushes.IndianRed,
                Title = "Red Axis"
            });
            cartesianChart1.AxisY.Add(new Axis
            {
                Foreground = System.Windows.Media.Brushes.DarkOliveGreen,
                Title = "Green Axis"
            });
        }
    }
}
