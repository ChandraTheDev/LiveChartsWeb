Sections

The cartesian chart supports sections, use a section to highlight or label values ranges

Code Behind

using System;
using System.Linq;
using System.Windows;
using System.Windows.Controls;
using System.Windows.Media;
using LiveCharts;
using LiveCharts.Defaults;
using LiveCharts.Wpf;
namespace Wpf.CartesianChart.Sections
{
    public partial class SectionsExample : UserControl
    {
        public SectionsExample()
        {
            InitializeComponent();
            SeriesCollection = new SeriesCollection
            {
                new LineSeries
                {
                    Values = new ChartValues<ObservableValue>
                    {
                        new ObservableValue(3),
                        new ObservableValue(5),
                        new ObservableValue(2),
                        new ObservableValue(7),
                        new ObservableValue(7),
                        new ObservableValue(4)
                    },
                    PointDiameter = 0,
                    StrokeThickness = 4,
                    Fill = Brushes.Transparent
                },
                new LineSeries
                {
                    Values = new ChartValues<ObservableValue>
                    {
                        new ObservableValue(3),
                        new ObservableValue(4),
                        new ObservableValue(6),
                        new ObservableValue(8),
                        new ObservableValue(7),
                        new ObservableValue(5)
                    },
                    PointDiameter = 0,
                    StrokeThickness = 4,
                    Fill = Brushes.Transparent
                }
            };
            DataContext = this;
        }
        public SeriesCollection SeriesCollection { get; set; }
    }
}

XAML

        <lvc:CartesianChart Grid.Row="2" Series="{Binding SeriesCollection}" >
            <lvc:CartesianChart.AxisY>
                <lvc:Axis>
                    <lvc:Axis.Sections>
                        <lvvc:AxisSection FromValue="8.5" ToValue="8.5" StrokeThickness="3" Stroke="#F9D648">
                            <vlc:AxisSection.Fill>
                                <SolidColorBrush Color="#A3A3FF" Opacity=".4"></SolidColorBrush>
                            </vlc:AxisSection.Fill>
                        </lvc:AxisSection>
                        <lvvc:AxisSection FromValue="4" ToValue="8" Label="Good">
                            <vlc:AxisSection.Fill>
                                <SolidColorBrush Color="#CDCDCD" Opacity=".4"></SolidColorBrush>
                            </vlc:AxisSection.Fill>
                        </lvc:AxisSection>
                        <lvc:AxisSection FromValue="0" ToValue="4" Label="Bad">
                            <lvc:AxisSection.Fill>
                                <SolidColorBrush Color="#FF8585" Opacity=".4"></SolidColorBrush>
                            </lvc:AxisSection.Fill>
                        </lc:AxisSection>
                    </lvc:Axis.Sections>
                </lc:Axis>
            </lvc:CartesianChart.AxisY>
        </lc:CartesianChart>
